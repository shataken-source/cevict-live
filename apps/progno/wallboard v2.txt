    if (data.type === 'odds_update') {
      handleOddsUpdate(data.payload);
    }

    if (data.type === 'prediction_update') {
      handlePredictionUpdate(data.payload);

    if (data.type === 'bet_result') {
      handleBetResult(data.payload);
    }
  };

  ws.onclose = () => {
    alertPush('red', 'Stream Disconnected', 'Reconnecting...');
    setTimeout(connectStream, 3000);
  };
}
