<script>
let games=[];
let picks=[];
let bankroll=[1000];

function renderGames(){
  const grid=document.getElementById("gamesGrid");
  grid.innerHTML="";

  games.forEach(g=>{
    const card=document.createElement("div");
    card.className="game-card";

    if(g.clockSeconds<300 && g.exposure) card.classList.add("zoom");

    const confidence=Math.abs(g.model-g.market);

    const ringColor=confidence>5?"var(--green)":confidence<2?"#333":"var(--yellow)";

    card.innerHTML=`
      <div class="confidence-ring" style="border-color:${ringColor}"></div>
      <div class="team-row">
        <div>${g.teamA}</div>
        <div>${g.teamB}</div>
      </div>
      <div class="score">${g.scoreA} - ${g.scoreB}</div>
      <div class="meta">${g.clock}</div>
      <div class="edge">Model: ${g.model}% | Market: ${g.market}% | Edge: ${(g.model-g.market).toFixed(1)}%</div>
    `;

    grid.appendChild(card);
  });
}

function renderHeatmap(){
  const container=document.getElementById("heatmapContainer");
  container.innerHTML="";
  picks.forEach(p=>{
    const edge=(p.model-p.market);
    let color=edge>3?"var(--green)":edge>1?"var(--yellow)":"var(--red)";
    const row=document.createElement("div");
    row.className="heat-row";
    row.style.color=color;
    row.innerHTML=`${p.label} | ${edge.toFixed(1)}%`;
    container.appendChild(row);
  });
}

function renderBankroll(){
  const canvas=document.getElementById("bankrollChart");
  const ctx=canvas.getContext("2d");
  canvas.width=canvas.offsetWidth;
  canvas.height=canvas.offsetHeight;
  ctx.clearRect(0,0,canvas.width,canvas.height);

  ctx.strokeStyle="#00ff9c";
  ctx.beginPath();
  bankroll.forEach((b,i)=>{
    const x=i*(canvas.width/(bankroll.length-1));
    const y=canvas.height - (b/Math.max(...bankroll))*canvas.height;
    if(i===0) ctx.moveTo(x,y);
    else ctx.lineTo(x,y);
  });
  ctx.stroke();
}

function renderBets(){
  const c=document.getElementById("betsContainer");
  c.innerHTML="";
  games.filter(g=>g.exposure).forEach(g=>{
    const div=document.createElement("div");
    div.innerHTML=`${g.teamA} vs ${g.teamB} | Exposure: $${g.exposure}`;
    c.appendChild(div);
  });
}

/* CALL THESE FROM YOUR DATA FEEDS */
function updateAll(newGames,newPicks,newBankroll){
  games=newGames;
  picks=newPicks;
  bankroll=newBankroll;
  renderGames();
  renderHeatmap();
  renderBankroll();
  renderBets();
}
